Port from `lib/bar.js` — iterate rows (0..height+2), for each datum compute ratio = height - (height * value / max), choose pad char based on ratio vs current row.

```rust
use crate::color::{Style, PAD};
use crate::datum::Datum;
use crate::options::BarOptions;

fn pad_mid(s: &str, width: usize) -> String {
    if s.len() >= width {
        s.chars().take(width).collect()
    } else {
        let total_pad = width - s.len();
        let left = total_pad / 2;
        let right = total_pad - left;
        format!("{}{}{}", " ".repeat(left), s, " ".repeat(right))
    }
}

pub fn bar(data: &[Datum], opts: &BarOptions) -> String {
    if data.is_empty() {
        return String::new();
    }

    let &BarOptions { bar_width, left, height, padding, style } = opts;
    let values: Vec<f64> = data.iter().map(|d| d.value_as_scalar().unwrap_or(0.0)).collect();
    let max = values.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
    if max <= 0.0 {
        return String::new();
    }

    let pad = " ";
    let mut result = pad.repeat(left);

    for i in 0..(height + 2) {
        for (j, item) in data.iter().iter().enumerate() {
            let val = values[j];
            let val_str = format_number(val);
            let ratio = height as f64 - (height as f64 * val / max);

            let pad_char = if ratio > (i as f64 + 2.0) {
                ' '
            } else if ratio.round() == i as f64 {
                // value label
                result.push_str(&pad_mid(&val_str, bar_width));
                result.push_str(&pad.repeat(padding));
                continue;
            } else if ratio < i as f64 {
                item.style.as_ref().map(|s| s.fg_char()).unwrap_or(style)
            } else {
                ' '
            };

            if i != height + 1 {
                let styled = apply_style_to_char(pad_char, &item.style);
                result.push_str(&styled.repeat(bar_width));
                result.push_str(&pad.repeat(padding));
            } else {
                let key_str = if item.key.len() > bar_width {
                    item.key.chars().take(bar_width + padding).collect::<String>()
                } else {
                    pad_mid(&item.key, bar_width) + &pad.repeat(padding)
                };
                result.push_str(&key_str);
            }
        }
        if i != height + 1 {
            result.push('
');
            result.push_str(&pad.repeat(left));
        }
    }

    result
}

fn format_number(v: f64) -> String {
    if v == v.floor() {
        format!("{:.0}", v)
    } else {
        format!("{}", v)
    }
}

fn apply_style_to_char(ch: char, style: &Option<Style>) -> String {
    match style {
        Some(s) => {
            if let Some((color, _)) = s.fg {
                format!("\x1b[{}m{}\x1b[0m", color.fg_code(), ch)
            } else {
                ch.to_string()
            }
        }
        None => ch.to_string(),
    }
}
```

Port from `lib/pie.js` — compute angles, render rows from -radius to +radius using circle equation.

```rust
use crate::datum::Datum;
use crate::options::PieOptions;

pub fn pie(data: &[Datum], opts: &PieOptions) -> String {
    if data.is_empty() {
        return String::new();
    }

    let total: f64 = data.iter().map(|d| d.value_as_scalar().unwrap_or(0.0)).sum();
    if total <= 0.0 {
        return String::new();
    }

    let radius = 5.0;
    let left = opts.left;
    let pad = " ";

    // Compute start/end angles for each slice
    let mut angles: Vec<(f64, f64, &Datum)> = Vec::new();
    let mut current = 0.0;
    for d in data {
        let val = d.value_as_scalar().unwrap_or(0.0);
        let sweep = (val / total) * 2.0 * std::f64::consts::PI;
        angles.push((current, current + sweep, d));
        current += sweep;
    }

    let mut result = String::new();
    let rows = (radius * 2.0 + 1.0) as i32;

    for r in -rows / 2..=rows / 2 {
        result.push_str(&pad.repeat(left));
        let y = r as f64;
        let x_max = (radius * radius - y * y).sqrt();

        for c in (-x_max.ceil() as i32)..=(x_max.ceil() as i32) {
            let x = c as f64;
            let dist_sq = x * x + y * y;
            if dist_sq <= radius * radius {
                let angle = (y).atan2(x);
                let normalized = if angle < 0.0 { angle + 2.0 * std::f64::consts::PI } else { angle };
                let slice = angles.iter().find(|(start, end, _)| {
                    normalized >= *start && normalized < *end
                });
                match slice {
                    Some((_, _, datum)) => {
                        let ch = datum.style.as_ref().and_then(|s| s.fg).map(|(_, ch)| ch).unwrap_or(' ');
                        let rendered = render_styled(ch, &datum.style);
                        result.push_str(&rendered);
                    }
                    None => result.push(' '),
                }
            } else {
                result.push(' ');
            }
        }
        result.push('
');
    }

    // Legend
    for (_, _, datum) in &angles {
        let ch = datum.style.as_ref().and_then(|s| s.fg).map(|(_, ch)| ch).unwrap_or(' ');
        result.push_str(&format!("{}: {} ({})
", ch, datum.key, datum.value_as_scalar().unwrap_or(0.0)));
    }

    result.trim_end().to_string()
}

fn render_styled(ch: char, style: &Option<crate::color::Style>) -> String {
    match style {
        Some(s) => {
            if let Some((color, _)) = s.fg {
                format!("\x1b[{}m{}\x1b[0m", color.fg_code(), ch)
            } else {
                ch.to_string()
            }
        }
        None => ch.to_string(),
    }
}
```

```rust
use crate::datum::Datum;
use crate::options::BulletOptions;

pub fn bullet(data: &[Datum], opts: &BulletOptions) -> String {
    if data.is_empty() {
        return String::new();
    }

    let max_val = data.iter()
        .map(|d| d.value_as_scalar().unwrap_or(0.0))
        .fold(f64::NEG_INFINITY, f64::max);
    if max_val <= 0.0 {
        return String::new();
    }

    let &BulletOptions { width, bar_width, style, height } = opts;
    let key_width = data.iter().map(|d| d.key.len()).max().unwrap_or(0);
    let bar_area = width.saturating_sub(key_width + 4);
    if bar_area == 0 {
        return String::new();
    }

    let mut result = String::new();

    for datum in data {
        let val = datum.value_as_scalar().unwrap_or(0.0);
        let filled = ((val / max_val) * bar_area as f64).round() as usize;
        let empty = bar_area - filled;

        let bar_char = datum.style.as_ref().and_then(|s| s.fg).map(|(_, ch)| ch).unwrap_or(style);
        let bar = render_bar(bar_char, filled, &datum.style);
        let empty_fill = " ".repeat(empty);

        result.push_str(&format!(
            "{:key_width$} |{}{}|\n",
            datum.key,
            bar,
            empty_fill
        ));
    }

    result.trim_end().to_string()
}

fn render_bar(ch: char, count: usize, style: &Option<crate::color::Style>) -> String {
    let s = ch.to_string().repeat(count);
    match style {
        Some(st) => {
            if let Some((color, _)) = st.fg {
                format!("\x1b[{}m{}\x1b[0m", color.fg_code(), s)
            } else if let Some((color, _)) = st.bg {
                format!("\x1b[{}m{}\x1b[0m", color.bg_code(), " ".repeat(count))
            } else {
                s
            }
        }
        None => s,
    }
}
```

```rust
use crate::datum::Datum;
use crate::options::DonutOptions;

pub fn donut(data: &[Datum], opts: &DonutOptions) -> String {
    if data.is_empty() {
        return String::new();
    }

    let total: f64 = data.iter().map(|d| d.value_as_scalar().unwrap_or(0.0)).sum();
    if total <= 0.0 {
        return String::new();
    }

    let outer_r = 5.0;
    let inner_r = outer_r / 2.0;
    let left = opts.left;

    let mut angles: Vec<(f64, f64, &Datum)> = Vec::new();
    let mut current = 0.0;
    for d in data {
        let val = d.value_as_scalar().unwrap_or(0.0);
        let sweep = (val / total) * 2.0 * std::f64::consts::PI;
        angles.push((current, current + sweep, d));
        current += sweep;
    }

    let rows = (outer_r * 2.0 + 1.0) as i32;
    let mut result = String::new();

    for r in -rows / 2..=rows / 2 {
        result.push_str(&" ".repeat(left));
        let y = r as f64;

        for c in -(outer_r.ceil() as i32)..=(outer_r.ceil() as i32) {
            let x = c as f64;
            let dist_sq = x * x + y * y;

            if dist_sq <= outer_r * outer_r && dist_sq >= inner_r * inner_r {
                let angle = y.atan2(x);
                let normalized = if angle < 0.0 { angle + 2.0 * std::f64::consts::PI } else { angle };
                let slice = angles.iter().find(|(s, e, _)| normalized >= *s && normalized < *e);
                match slice {
                    Some((_, _, datum)) => {
                        let ch = datum.style.as_ref().and_then(|s| s.fg).map(|(_, ch)| ch).unwrap_or(' ');
                        result.push_str(&render_styled(ch, &datum.style));
                    }
                    None => result.push(' '),
                }
            } else {
                result.push(' ');
            }
        }
        result.push('
');
    }

    result.trim_end().to_string()
}

fn render_styled(ch: char, style: &Option<crate::color::Style>) -> String {
    match style {
        Some(s) => {
            if let Some((color, _)) = s.fg {
                format!("\x1b[{}m{}\x1b[0m", color.fg_code(), ch)
            } else {
                ch.to_string()
            }
        }
        None => ch.to_string(),
    }
}
```

```rust
use crate::datum::Datum;
use crate::options::GaugeOptions;

pub fn gauge(data: &[Datum], opts: &GaugeOptions) -> String {
    if data.is_empty() {
        return String::new();
    }

    let radius = opts.radius as f64;
    let datum = &data[0];
    let val = datum.value_as_scalar().unwrap_or(0.0).clamp(0.0, 1.0);

    let rows = (radius * 2.0 + 1.0) as i32;
    let mut result = String::new();

    for r in 0..=(rows / 2) {
        let y = (rows / 2 - r) as f64;
        let x_max = (radius * radius - y * y).sqrt();

        for c in -(x_max.ceil() as i32)..=(x_max.ceil() as i32) {
            let x = c as f64;
            let angle = y.atan2(-x);
            let normalized = if angle < 0.0 { angle + 2.0 * std::f64::consts::PI } else { angle };

            if normalized <= std::f64::consts::PI {
                let filled = normalized / std::f64::consts::PI <= val;
                let ch = if filled {
                    datum.style.as_ref().and_then(|s| s.fg).map(|(_, ch)| ch).unwrap_or('█')
                } else {
                    ' '
                };
                result.push_str(&render_styled(ch, &datum.style));
            } else {
                result.push(' ');
            }
        }
        result.push('
');
    }

    // Label
    if opts.show_percentage {
        result.push_str(&format!("{}: {:.0}%\n", datum.key, val * 100.0));
    }

    result.trim_end().to_string()
}

fn render_styled(ch: char, style: &Option<crate::color::Style>) -> String {
    match style {
        Some(s) => {
            if let Some((color, _)) = s.fg {
                format!("\x1b[{}m{}\x1b[0m", color.fg_code(), ch)
            } else {
                ch.to_string()
            }
        }
        None => ch.to_string(),
    }
}
```

```rust
use crate::datum::Datum;
use crate::options::ScatterOptions;

pub fn scatter(data: &[Datum], opts: &ScatterOptions) -> String {
    if data.is_empty() {
        return String::new();
    }

    let width = opts.width as usize;
    let height = width; // square grid

    // Find bounds
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for d in data {
        if let crate::DatumValue::Point([x, y]) = &d.value {
            min_x = min_x.min(*x);
            max_x = max_x.max(*x);
            min_y = min_y.min(*y);
            max_y = max_y.max(*y);
        }
    }

    if max_x <= min_x && max_y <= min_y {
        return String::new();
    }

    let x_range = (max_x - min_x).max(1.0);
    let y_range = (max_y - min_y).max(1.0);

    // Create grid
    let mut grid = vec![vec![' '; width]; height];

    for d in data {
        if let crate::DatumValue::Point([x, y]) = &d.value {
            let col = ((x - min_x) / x_range * (width - 1) as f64).round() as usize;
            let row = ((max_y - y) / y_range * (height - 1) as f64).round() as usize;
            let col = col.min(width - 1);
            let row = row.min(height - 1);

            let ch = d.style.as_ref().and_then(|s| s.fg).map(|(_, ch)| ch).unwrap_or('*');
            grid[row][col] = ch;
        }
    }

    // Render
    let mut result = String::new();
    for row in &grid {
        for &ch in row {
            result.push(ch);
        }
        result.push('
');
    }

    // Legend
    for d in data {
        if let crate::DatumValue::Point([x, y]) = &d.value {
            result.push_str(&format!("{}: ({}, {})\n", d.key, x, y));
        }
    }

    result.trim_end().to_string()
}
```

```rust
fn parse_color(s: &str) -> Color {
    match s.to_lowercase().as_str() {
        "black" => Color::Black,
        "red" => Color::Red,
        "green" => Color::Green,
        "yellow" => Color::Yellow,
        "blue" => Color::Blue,
        "magenta" => Color::Magenta,
        "cyan" => Color::Cyan,
        "white" => Color::White,
        _ => Color::White,
    }
}

fn load_data(data: Option<&str>, file: Option<&str>) -> Vec<Datum> {
    let json = match (data, file) {
        (Some(d), _) => d.clone(),
        (_, Some(f)) => std::fs::read_to_string(f).expect("Failed to read data file"),
        (None, None) => {
            let mut buf = String::new();
            std::io::stdin().read_to_string(&mut buf).ok();
            buf
        }
    };
    serde_json::from_str(&json).expect("Invalid JSON data")
}

        Commands::Donut { data, data_file, left } => {
            let data = load_data(data.as_deref(), data_file.as_deref());
            let opts = DonutOptions { left, ..Default::default() };
            println!("{}", ervy::donut(&data, &opts));
        }
        Commands::Gauge { data, data_file, radius } => {
            let data = load_data(data.as_deref(), data_file.as_deref());
            let opts = GaugeOptions {
                radius,
                ..Default::default()
            };
            println!("{}", ervy::gauge(&data, &opts));
        }
        Commands::Scatter { data, data_file, width, legend_gap } => {

        }
        Commands::Bg { color, length } => {
            let c = parse_color(&color);
            print!("{}", bg(c, length).render_bg(length));
        }
    }
}
```
